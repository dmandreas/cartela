\chapter{KOMMUNIONUTDELNING TILL DE SJUKA}
\markboth{KOMMUNIONUTDELNING}{}


\rub{Accedens ad infirmum, sacerdos ipsum ceterosque astantes humaniter salutat, adhibens salutationem:}

\begin{multicols}{2}
\renewcommand{\columnseprulecolor}{\color{rubrica}}
\renewcommand{\columnseprule}{0.4pt}
\va In nómine Patris et Fílii et Spíritus Sancti.\\
\ra Amen\\
\va
Pax huic dómui
et ómnibus habitántibus in ea.
\columnbreak

\va I Faderns och Sonens och den helige Andes namn.\\
\ra Amen\\
\va Frid vare över detta hus och alla som bor häri.
\end{multicols}


\medskip
\rub{Tunc, deposito Sacramento super mensam, ipsum cum astantibus adorat. Deinde,
pro opportunitate, accepta aqua benedicta, aspergit infirmum et cubiculum, dicens:}

\medskip

\begin{multicols}{2}
\renewcommand{\columnseprulecolor}{\color{rubrica}}
\renewcommand{\columnseprule}{0.4pt}
Sit hæc aqua suscépti baptísmatis memória et Christum récolat, qui
passióne et resurrectióne sua nos redémit.
\columnbreak

Må detta vatten påminna dig och oss om vårt dop till 
 frälsning i Kristi död och uppståndelse.
\end{multicols}

\medskip
\rub{Sacerdos infirmum ceterosque astantes invitat ad actum pænitentialem dicendum:}

\begin{multicols}{2}
\renewcommand{\columnseprulecolor}{\color{rubrica}}
\renewcommand{\columnseprule}{0.4pt}
\va Fratres, agnoscámus peccáta nostra, ut apti simus ad hanc sacram
celebratiónem participándam.
\columnbreak

\va Låt oss besinna oss och bekänna våra synder så att vi blir väl beredda att fira denna heliga handling.
\end{multicols}

\medskip
\rub{Et fit brevis pausa silentii. Postea omnes simul faciunt confessionem:}
\smallskip

\PC{Confíteor Deo omnipoténti et vobis, fratres, quia peccávi nimis cogitatióne,
verbo, ópere et omissióne: mea culpa, mea culpa, mea
máxima culpa. Ideo precor Beátam Maríam semper Vírginem, omnes
ángelos et sanctos, et vos, fratres, oráre pro me ad Dóminum Deum
nostrum.}
{Jag bekänner inför Gud alls\-mäk\-tig och er, mina brö\-der och sys\-trar, att jag har syndat i tankar och ord, gär\-nin\-gar och underlåtelse, genom min skuld, min skuld, min egen stora skuld. Därför ber jag den saliga, alltid rena jungfrun Maria, alla änglar och helgon och er, mina bröder och sys\-trar, att be för mig till Herren, vår Gud.\par}


\smallskip
\rub{Et sacerdos concludit:}

\begin{multicols}{2}
\renewcommand{\columnseprulecolor}{\color{rubrica}}
\renewcommand{\columnseprule}{0.4pt}

\va Misereátur nostri omnípotens Deus, et, dimíssis peccátis nostris, perdúcat
nos ad vitam ætérnam.\\
\ra Amen.
\columnbreak

\va Gud allsmäktig förlåte oss våra synder i sin stora barmhärtighet och före oss till det eviga livet.\\
\ra Amen.
\end{multicols}

\smallskip
\rub{Tunc, pro opportunitate, ab uno ex astantibus vel ab ipso sacerdote legi potest textus
Sacræ Scripturæ, v. g. Io 6, 54-55:}
\smallskip

\begin{multicols}{2}
\renewcommand{\columnseprulecolor}{\color{rubrica}}
\renewcommand{\columnseprule}{0.4pt}
Qui mandúcat meam carnem et bibit meum sánguinem habet vitam
ætérnam, et ego resuscitábo eum in novíssimo die. Caro enim mea
verus est cibus, et sanguis meus verus est potus.
\columnbreak

Den som äter mitt kött och dricker mitt blod har evigt liv, och jag ska låta honom uppstå på den sista dagen. Ty mitt kött är verklig föda, och mitt blod är verklig dryck.
\end{multicols}

\smallskip
\rub{Tunc sacerdos his verbis recitationem orationis dominicæ introducit:}

\begin{multicols}{2}
\renewcommand{\columnseprulecolor}{\color{rubrica}}
\renewcommand{\columnseprule}{0.4pt}
\va Nunc autem una simul Deum deprecémur, sicut Dóminus noster
Iesus Christus nos dócuit oráre:
\columnbreak

\va Låt oss nu be den bön, som vår Herre Jesus Kristus har lärt oss:
\end{multicols}

\medskip
\rub{et omnes simul pergunt:}

\begin{multicols}{2}
\renewcommand{\columnseprulecolor}{\color{rubrica}}
\renewcommand{\columnseprule}{0.4pt}
Pater noster\ldots\\
Fader vår\ldots
\end{multicols}

\medskip
\rub{Tunc sacerdos ostendit Ss.mum Sacramentum, dicens:}
\smallskip

\PC{\va Ecce Agnus Dei, ecce qui tollit peccáta mundi. Beáti qui ad cenam Agni vocáti sunt.}{\va Se Guds lamm, som borttager världens synder. Saliga de som blivit kallade till Lammets måltid.}

\PC{\ra Dómine, non sum dignus ut intres sub tectum meum: sed tantum dic verbo, et sanábitur ánima mea.}{\ra Herre, jag är inte värdig att du går in under mitt tak, men säg bara ett ord, så blir min själ helad.}


\medskip
\rub{Sacerdos ad infirmum accedit, eique Sacramentum ostendens, dicit:}

\begin{multicols}{2}
\renewcommand{\columnseprulecolor}{\color{rubrica}}
\renewcommand{\columnseprule}{0.4pt}
\va Corpus Christi.\\
\ra Amen.
\columnbreak

\va Kristi kropp.\\
\ra Amen.
\end{multicols}

\medskip
\rub{Distributione Communionis expleta, purificatio de more peragitur. Deinde sacerdos
dicit orationem conclusivam:}

\PC{Orémus.}{Låt oss bedja.}

\PC{Dómine, sancte Pater, omnípotens ætérne Deus, te fidéliter deprecámur,
ut sacrosánctum Corpus Dómini nostri Iesu Christi
Fílii tui accipiénti fratri nostro {\color{rubrica}(}soróri nostræ{\color{rubrica})} tam córpori
quam ánimæ prosit ad remédium sempitérnum. Per Christum Dóminum
nostrum.}{Herre, barmhärtighetens \  Fader, med det himmelska brödet har du mättat vår hunger. Låt Sakramentet ge styrka för vårt liv här i tiden och göra oss värdiga det eviga. Genom Jesus Kristus vår Herre.}

\PC{\ra Amen.}{\ra Amen.}

\smallskip
\rub{Om kommunionen har utdelats till en döende:}
\smallskip

\PC{}{Herre, du är den eviga saligheten för alla som tror på dig. Låt din tjänare / tjänarinna {\color{rubrica}N.}, som har mättats med himmelsk mat och dryck, i frid och trygghet bryta upp från detta livet och gå in i ljusets och livets rike. Genom Jesus Kristus, vår Herre.\\
\ra Amen}


\smallskip
\rub{Deinde infirmo et astantibus benedicit: }
\smallskip

\begin{multicols}{2}
\renewcommand{\columnseprulecolor}{\color{rubrica}}
\renewcommand{\columnseprule}{0.4pt}
\va Benedícat vos omnípotens Deus, Pater, et Fílius, {\color{rubrica} $\maltese$} et Spíritus Sanctus.\\
\ra Amen.
\columnbreak

\va Välsigne er Gud allsmäktig, Fadern och Sonen {\color{rubrica} $\maltese$} och den helige Ande.\\
\ra Amen.
\end{multicols}

\renewcommand{\columnseprulecolor}{\color{rubrica}}

\columnbreak

\va Låt oss nu be den bön, som vår Herre Jesus Kristus har lärt oss:

\smallskip
\rub{Tunc sacerdos genuflectit, accipit hostiam, eamque parum elevatam super pyxidem
tenens, ad communicandos versus, dicit:}

\PC{
\va Ecce Agnus Dei, ecce qui tollit peccáta mundi. Beáti qui ad cenam Agni vocáti sunt.}{\va Se Guds lamm, som borttager världens synder. Saliga de som blivit kallade till Lammets måltid.}
\PC{\ra Dómine, non sum dignus ut intres sub tectum meum: sed tantum dic verbo, et sanábitur ánima mea.}{\ra Herre, jag är inte värdig att du går in under mitt tak, men säg bara ett ord, så blir min själ helad.}


\smallskip
\rub{Postea accipit pyxidem, accedit ad communicandos, et hostiam parum elevatam
unicuique eorum ostendit, dicens:}

\begin{multicols}{2}
\renewcommand{\columnseprulecolor}{\color{rubrica}}
\renewcommand{\columnseprule}{0.4pt}

\ra Amen.
\columnbreak

\va Kristi kropp.\\
\ra Amen.
\end{multicols}

\smallskip
\rub{Distributione Communionis expleta, purificatio de more peragitur. Deinde sacerdos
dicit orationem conclusivam:}


\begin{multicols}{2}
\renewcommand{\columnseprulecolor}{\color{rubrica}}
\renewcommand{\columnseprule}{0.4pt}
Orémus.\\
Deus, qui nobis sub sacraménto mirábili passiónis tuæ memóriam
reliquísti, tríbue, qu\'\ae sumus, ita nos Córporis et Sánguinis tui
sacra mystéria venerári, ut redemptiónis tuæ fructum in nobis
iúgiter sentiámus. Qui vivis et regnas in s\' \ae cula sæculórum.\\
\ra Amen.

\columnbreak

Låt oss bedja.\\
Herre Jesus Kristus, vi tackar dig för att du i Altarets Sakrament har lämnat oss en åminnelse av ditt lidande. Lär oss att så vörda din kropps och ditt blods mysterium att vi i vårt eget liv får erfara din frälsning. Du som lever och råder i evigheters evighet.\\
\ra Amen

\end{multicols}

\smallskip
